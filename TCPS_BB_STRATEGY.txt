//@version=6
strategy("TCPS BB Mean Reversion v2", overlay=true, initial_capital=10000,
  default_qty_type=strategy.percent_of_equity, default_qty_value=100,
  commission_type=strategy.commission.percent, commission_value=0.1)

// Bollinger Bands + EMA filter
bbLen = input.int(20, "BB Length")
bbStd = input.float(2.0, "BB StdDev")
emaLen = input.int(50, "Trend EMA")
slPct = input.float(0.8, "Stop Loss %")
tpPct = input.float(1.5, "Take Profit %")

[bbMid, bbUp, bbDown] = ta.bb(close, bbLen, bbStd)
emaVal = ta.ema(close, emaLen)

plot(bbMid, "Mid", color.blue)
plot(bbUp, "Upper", color.gray, 1, style=plot.style_circles)
plot(bbDown, "Lower", color.gray, 1, style=plot.style_circles)
plot(emaVal, "EMA50", color.orange)

// Entry: BB touch + EMA direction filter
longCond = low <= bbDown and emaVal < emaVal[1]
shortCond = high >= bbUp and emaVal > emaVal[1]
cooldown = 3

var int lastBar = 0

if bar_index - lastBar > cooldown
    if longCond
        strategy.entry("L", strategy.long)
        lastBar := bar_index
    if shortCond
        strategy.entry("S", strategy.short)
        lastBar := bar_index

strategy.exit("XL", "L", loss=close*slPct/100, profit=close*tpPct/100)
strategy.exit("XS", "S", loss=close*slPct/100, profit=close*tpPct/100)
