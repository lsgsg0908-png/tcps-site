//@version=6
indicator(title="TCPS Structure Mapper [FREE]", shorttitle="TCPS Structure", overlay=true)

// ===== 무료 제공용 지표 =====
// 간단한 구조 분석 + 주요 지지/저항

// Inputs
lookback = input.int(10, "Swing Lookback", minval=5)
showLevels = input.bool(true, "Show S/R Levels")

// Swing Highs/Lows
swHigh = ta.pivothigh(high, lookback, lookback)
swLow = ta.pivotlow(low, lookback, lookback)

// Plot swings
plotshape(swHigh, style=shape.labeldown, location=location.abovebar,
          color=color.new(#ff6b6b, 0), textcolor=color.white,
          size=size.tiny, text="H")

plotshape(swLow, style=shape.labelup, location=location.belowbar,
          color=color.new(#51cf66, 0), textcolor=color.white,
          size=size.tiny, text="L")

// Key levels (last 10 swings)
var float[] levelsHigh = array.new<float>(0)
var float[] levelsLow = array.new<float>(0)

if not na(swHigh)
    array.push(levelsHigh, swHigh)
    if array.size(levelsHigh) > 10
        array.shift(levelsHigh)

if not na(swLow)
    array.push(levelsLow, swLow)
    if array.size(levelsLow) > 10
        array.shift(levelsLow)

// Draw key levels
if showLevels
    for i = 0 to array.size(levelsHigh) - 1
        line.new(bar_index - 20, array.get(levelsHigh, i),
                 bar_index, array.get(levelsHigh, i),
                 color=color.new(#ff6b6b, 70), width=1,
                 style=line.style_dashed, extend=extend.right)

    for i = 0 to array.size(levelsLow) - 1
        line.new(bar_index - 20, array.get(levelsLow, i),
                 bar_index, array.get(levelsLow, i),
                 color=color.new(#51cf66, 70), width=1,
                 style=line.style_dashed, extend=extend.right)

// Market structure label
float lastH = na, float lastL = na
for i = 0 to array.size(levelsHigh) - 1
    lastH := array.get(levelsHigh, 0)
for i = 0 to array.size(levelsLow) - 1
    lastL := array.get(levelsLow, 0)

bool uptrend = not na(lastL) and not na(lastH) and lastL > ta.valuewhen(swLow, swLow, 1)
bool downtrend = not na(lastL) and not na(lastH) and lastH < ta.valuewhen(swHigh, swHigh, 1)

if uptrend
    label.new(bar_index, low - (high - low) * 2,
              "▲ UPTREND", style=label.style_none,
              color=color.new(#51cf66, 80), textcolor=#51cf66,
              size=size.small)

if downtrend
    label.new(bar_index, high + (high - low) * 2,
              "▼ DOWNTREND", style=label.style_none,
              color=color.new(#ff6b6b, 80), textcolor=#ff6b6b,
              size=size.small)

// Alert
alertcondition(uptrend or downtrend, title="Structure Change",
               message="Market structure changed: {{ticker}}")
