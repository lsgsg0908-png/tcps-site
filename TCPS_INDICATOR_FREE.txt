//@version=6
indicator("TCPS Structure Mapper", overlay=true, max_labels_count=50)

// FREE indicator: Simple swing structure analysis
swLen = input.int(10, "Swing Length")

ph = ta.pivothigh(high, swLen, swLen)
pl = ta.pivotlow(low, swLen, swLen)

plotshape(ph, "PH", style=shape.triangledown, location=location.abovebar, color=color.red, size=size.small)
plotshape(pl, "PL", style=shape.triangleup, location=location.belowbar, color=color.green, size=size.small)

// Nearest levels
var float r1 = na
var float s1 = na
if not na(ph)
    r1 := ph
    line.new(bar_index, ph, bar_index + 10, ph, color=color.new(color.red, 70), extend=extend.right, style=line.style_dashed)
if not na(pl)
    s1 := pl
    line.new(bar_index, pl, bar_index + 10, pl, color=color.new(color.green, 70), extend=extend.right, style=line.style_dashed)

// Structure bias
bool uptrend = not na(pl) and not na(ph) and ph > ph[1]
bool downtrend = not na(pl) and not na(ph) and ph < ph[1]

plotchar(uptrend, "Up", "▲", location.belowbar, color.green, size=size.tiny)
plotchar(downtrend, "Dn", "▼", location.abovebar, color.red, size=size.tiny)
