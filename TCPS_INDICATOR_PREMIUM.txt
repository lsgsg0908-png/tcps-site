//@version=6
indicator(title="TCPS Institutional Flow Detector [PREMIUM]", shorttitle="TCPS IFD", overlay=true)

// ===== 유료 판매용 프리미엄 지표 =====
// 3-in-1: CHOCH + SMC OB + Liquidity Sweep 통합

// ─── Inputs ───
swingLen = input.int(5, "Swing Length")
emaF = input.int(20, "EMA Fast")
emaS = input.int(50, "EMA Slow")
sweepLen = input.int(1, "Sweep Lookback")
obStrength = input.int(60, "OB Strength Filter")
showSignals = input.bool(true, "Show Entry Signals")

// ─── EMA ───
eFast = ta.ema(close, emaF)
eSlow = ta.ema(close, emaS)

// ─── CHOCH (Structure Shift) ───
ph = ta.pivothigh(high, swingLen, swingLen)
pl = ta.pivotlow(low, swingLen, swingLen)

var float lastPH = na
var float lastPL = na
var string bias = "neutral"
var int lastSigBar = 0

if not na(ph) and ph > lastPH
    lastPH := ph
if not na(pl) and pl < lastPL
    lastPL := pl

if not na(lastPH) and close > lastPH and bias != "bull"
    bias := "bull"
    if showSignals and bar_index - lastSigBar > 5
        label.new(bar_index, low, "▲ CHOCH", style=label.style_label_up,
                  color=color.new(#2196F3, 80), textcolor=color.white, size=size.small)
        lastSigBar := bar_index

if not na(lastPL) and close < lastPL and bias != "bear"
    bias := "bear"
    if showSignals and bar_index - lastSigBar > 5
        label.new(bar_index, high, "▼ CHOCH", style=label.style_label_down,
                  color=color.new(#FF9800, 80), textcolor=color.white, size=size.small)
        lastSigBar := bar_index

// ─── SMC Order Block ───
body = math.abs(close - open)
cRange = high - low
bodyScore = (1 - body / cRange) * 100

bullOB = eFast > eSlow and close[1] < open[1] and close > close[1] and bodyScore > obStrength
bearOB = eFast < eSlow and close[1] > open[1] and close < close[1] and bodyScore > obStrength

if bullOB
    box.new(bar_index, high[1], bar_index + 30, low[1],
            bgcolor=color.new(#4CAF50, 85), border_color=color.new(#4CAF50, 60))

if bearOB
    box.new(bar_index, high[1], bar_index + 30, low[1],
            bgcolor=color.new(#f44336, 85), border_color=color.new(#f44336, 60))

// ─── Liquidity Sweep ───
lowestBarrier = ta.lowest(low[1], sweepLen)
highestBarrier = ta.highest(high[1], sweepLen)

sweepLong = close[1] < open[1] and close > open and low < lowestBarrier
sweepShort = close[1] > open[1] and close < open and high > highestBarrier

if sweepLong
    label.new(bar_index, low, "⚡SWEEP", style=label.style_label_up,
              color=color.new(#00BCD4, 80), textcolor=color.white, size=size.tiny)

if sweepShort
    label.new(bar_index, high, "⚡SWEEP", style=label.style_label_down,
              color=color.new(#E91E63, 80), textcolor=color.white, size=size.tiny)

// ─── Composite Signal ───
bool longSignal = sweepLong and bias == "bull"
bool shortSignal = sweepShort and bias == "bear"
bool strongLong = longSignal and bullOB
bool strongShort = shortSignal and bearOB

// Big signals for strong entries
if strongLong
    label.new(bar_index, low - (high - low) * 1.5,
              "🟢 TCPS BUY", style=label.style_label_up,
              color=color.new(#00E676, 0), textcolor=color.white, size=size.normal)

if strongShort
    label.new(bar_index, high + (high - low) * 1.5,
              "🔴 TCPS SELL", style=label.style_label_down,
              color=color.new(#FF1744, 0), textcolor=color.white, size=size.normal)

// ─── Alerts ───
alertcondition(strongLong, title="TCPS Strong Buy",
               message="🟢 TCPS STRONG BUY: {{ticker}} @ {{close}}")
alertcondition(strongShort, title="TCPS Strong Sell",
               message="🔴 TCPS STRONG SELL: {{ticker}} @ {{close}}")
alertcondition(longSignal, title="TCPS Buy Signal",
               message="🟢 TCPS BUY: {{ticker}} @ {{close}}")
alertcondition(shortSignal, title="TCPS Sell Signal",
               message="🔴 TCPS SELL: {{ticker}} @ {{close}}")
